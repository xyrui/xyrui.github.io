For循环

\def\xunhuan{{a,b},{c,d}} % 定义宏
\foreach[count=\i] \x in \xunhuan  %count用于计数
{
    \foreach[count=\j] \y in \x
    {
        position: (\i,\j), context: \y \\
    }
}
